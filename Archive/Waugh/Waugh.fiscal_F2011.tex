\begin{frame}[t]
\frametitle{Taxes and Debt}
\bigskip
\begin{itemize}
\item Government budget:
\begin{eqnarray*}
G_t + V_t + i_tB_t = T_t + B_{t+1} - B_{t}
\end{eqnarray*}
\medskip
\item Ingredients:
\begin{itemize}
\medskip
\item $G =$ government purchases of goods and services
\medskip
\item $V =$ transfer payments from government to households
\medskip
\item $G + V =$ government spending
\medskip
\item $T =$ tax revenue
\medskip
\item $D = G + V - T =$ primary deficit (excl. interest)
\medskip
\item $B =$ government debt (``bonds'') at begining of period.
\medskip
\item $i =$ (nominal) interest rate paid on debt
\medskip
\item $iB =$ interest payments
\bigskip
\end{itemize}
\end{itemize}
\bigskip
\end{frame}

\begin{frame}[t]
\frametitle{How Should a Deficits Be Financed?}
\bigskip
\begin{itemize}
\item Government budget:
\begin{eqnarray*}
G_t + V_t + i_tB_t = T_t + B_{t+1} - B_{t}
\end{eqnarray*}
\medskip
\item If $G_t + V_t + i_tB_t - T_t > 0$
\begin{itemize}
\medskip
\item Raise taxes: increase $T_t$
\medskip
\item Issue debt: increase $B_t$
\end{itemize}
\bigskip
\item Is there a difference?
\end{itemize}
\bigskip
\end{frame}

\begin{frame}[t]
\frametitle{Taxes and Debt}
\bigskip
\begin{itemize}
\item Government budget constraint
\begin{eqnarray*}
G_t + V_t + i_tB_t = T_t + B_{t+1} - B_{t}
\end{eqnarray*}
\item Rewrite as
\begin{eqnarray*}
B_t(1+i_t) = B_{t+1} + T_t - G_t - V_t\\
\\
B_t(1+i_t) = B_{t+1} - D_t\\
\\
B_{t} = \frac{B_{t+1}}{1+i_{t}} - \frac{D_t}{1+i_{t}}
\end{eqnarray*}
\medskip
\item Debt must be backed by PV of tommorow's debt plus primary surplus
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[t]
\frametitle{Taxes and Debt: Present Value}
\bigskip
\begin{itemize}
\item Repeated substitution gives us
\begin{eqnarray*}
B_{t} = \frac{B_{t+n}}{(1+i)^n} - \bigg[\frac{D_{t}}{1+i} + \frac{D_{t+1}}{(1+i)^2} + \frac{D_{t+1}}{(1+i)^3} \ldots \frac{D_{t+n-1}}{(1+i)^n} \bigg]
\end{eqnarray*}
\medskip
\item In words, debt today is backed by value of future \textbf{primary} surpluses
\begin{itemize}
\medskip
\item Assumes $\frac{B_{t+n}}{(1+i)^n} \rightarrow 0$
\medskip
\item Don't need $B_{t+n}  \rightarrow 0$
\end{itemize}
\end{itemize}
\bigskip
\end{frame}

\begin{frame}[t]
\frametitle{Sustainability Analysis}
\bigskip
\begin{itemize}
\item Here is the issue:
\begin{itemize}
\medskip
\item What happens to ratio of debt to GDP if policy doesn’t change?
\medskip
\item It decreases/constant:  sustainable.  If not, unsustainable
\medskip
\item Both measured at current prices (“nominal”)
\bigskip
\end{itemize}
\item Growth of (nominal) debt:  $B_{t+1} = (1+i)B_{t} + D_{t} $
\bigskip
\item Growth of (nominal) GDP:  $Y_{t+1} = (1+g)Y_{t}$
\bigskip
\item Growth of debt to GDP ratio:  $\frac{B_{t+1}}{Y_{t+1}} = \frac{(1+i)B_t}{(1+g)Y_{t}} + \frac{D_t}{(1+g)Y_{t}}$
\end{itemize}
\bigskip
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}[t]
\frametitle{Sustainability Analysis}
\bigskip
\begin{itemize}
\item Long run ``natural'' growth rates
\begin{itemize}
\medskip
\item Debt: $(1+i)$
\medskip
\item Nominal GDP: $(1+g)$
\medskip
\item Ratio of dept to GDP: $\frac{1+i}{1+g}$
\bigskip
\end{itemize}
\item Sustainability requires $\frac{1+i}{1+g} < 1$ or $i < g$
\bigskip
\item Typically $i > g$, thus something must change---but what?
\end{itemize}
\bigskip
\end{frame}
